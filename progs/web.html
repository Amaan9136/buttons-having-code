                    <h1>Simple Calculator</h1>
                    <script>
                    const operand1 = parseFloat(prompt('Enter the first number'));
                    if (isNaN(operand1)) {
                        alert('Invalid input');
                    } else {
                        const operator = prompt('Enter operator (+, -, *, /)');
                        if (!['+', '-', '*', '/'].includes(operator)) {
                        alert('Invalid operator');
                        } else {
                        const operand2 = parseFloat(prompt('Enter the second number'));
                        if (isNaN(operand2)) {
                            alert('Invalid input');
                        } else {
                            let result;
                            switch (operator) {
                            case '+':
                                result = operand1 + operand2;
                                break;
                            case '-':
                                result = operand1 - operand2;
                                break;
                            case '*':
                                result = operand1 * operand2;
                                break;
                            case '/':
                                if (operand2 === 0) {
                                alert('Cannot divide by zero');
                                } else {
                                result = operand1 / operand2;
                                }
                                break;
                            }
                            alert('Result: ' + result);
                        }
                        }
                    }
                    </script>
                
            
        
            
                
                    <h1>Square and Cube</h1>
                    <div id="table"></div>
                    <script>
                        function square(num) {
                            num = num * num;
                            return num;
                        }

                        function cube(num) {
                            num = num * num * num;
                            return num;
                        }

                        const startNumber = parseInt(prompt("Enter the start number:"));
                        const endNumber = parseInt(prompt("Enter the end number:"));

                        let tableContent = '<table border="1">';
                        tableContent += '<tr><th>Number</th><th>Square</th><th>Cube</th></tr>';

                        if (!isNaN(startNumber) && !isNaN(endNumber)) {
                            for (let i = startNumber; i <= endNumber; i++) {
                                const squareResult = square(i);
                                const cubeResult = cube(i);
                                tableContent += `<tr><td>${i}</td><td>${squareResult}</td><td>${cubeResult}</td></tr>`;
                            }
                        } else {
                            alert('Please enter valid numbers');
                        }

                        tableContent += '</table>';

                        document.getElementById('table').innerHTML = tableContent;
                    </script>

                    <h1>Pattern Generator</h1>
                    <style>
                        body {
                            text-align: center;
                        }
                    </style>

                    <body>
                        <label for="patternSelect">Select a pattern:</label>
                        <select id="patternSelect">
                            <option value="square">Square</option>
                            <option value="leftTriangle">Left Triangle</option>
                            <option value="downwardTriangle">Downward Triangle</option>
                            <option value="pyramid">Pyramid</option>
                            <option value="diamond">Diamond</option>
                        </select>

                        <label for="rowsInput">Number of rows:</label>
                        <input type="number" id="rowsInput" value="4" min="1">

                        <button onclick="generateAndDisplayPattern()">Generate Pattern</button>

                        <div class="pattern" id="patternOutput"></div>

                        <script>
                            function generatePattern(rows, patternType) {
                                let pattern = '';

                                switch (patternType) {
                                    case 'square':
                                        for (let i = 0; i < rows; i++) {
                                            pattern += '*'.repeat(rows) + '<br>';
                                        }
                                        break;

                                    case 'downwardTriangle':
                                        for (let i = rows; i > 0; i--) {
                                            pattern += ' '.repeat(rows - i) + '*'.repeat(i) + '<br>';
                                        }
                                        break;

                                    case 'leftTriangle':
                                        for (let i = 0; i < rows; i++) {
                                            pattern += ' '.repeat(rows - i - 1) + '*'.repeat(i + 1) + '<br>';
                                        }
                                        break;

                                    case 'pyramid':
                                        for (let i = 0; i < rows; i++) {
                                            pattern += ' '.repeat(rows - i - 1) + '*'.repeat(2 * i + 1) + '<br>';
                                        }
                                        break;

                                    case 'diamond':
                                        for (let i = 0; i < rows; i++) {
                                            pattern += ' '.repeat(rows - i - 1) + '*'.repeat(2 * i + 1) + '<br>';
                                        }
                                        for (let i = rows - 2; i >= 0; i--) {
                                            pattern += ' '.repeat(rows - i - 1) + '*'.repeat(2 * i + 1) + '<br>';
                                        }
                                        break;

                                    default:
                                        pattern = 'Invalid pattern type.';
                                        break;
                                }

                                return pattern;
                            }

                            function generateAndDisplayPattern() {
                                const patternSelect = document.getElementById('patternSelect');
                                const rowsInput = document.getElementById('rowsInput');
                                const patternOutput = document.getElementById('patternOutput');

                                const patternType = patternSelect.value;
                                const rows = parseInt(rowsInput.value, 10);

                                const pattern = generatePattern(rows, patternType);
                                patternOutput.innerHTML = pattern;
                            }
                        </script>
                    </body>
